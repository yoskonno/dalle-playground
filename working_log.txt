    global announce_url

def start_cloudflared(port):
  from flask_cloudflared import _run_cloudflared
  try:
    announce_url = _run_cloudflared(port)
  except:
    raise
  finally:
    update_announce_url(announce_url)
    cloudflared_startup.set()

        new_thread.setDaemon(True)

        old_run(*args, **kwargs)
